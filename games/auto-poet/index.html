string rand_(string *t){
        int k,cnt = 0;
        while(t[cnt] != "") ++cnt;
        do{
                k = rand() % cnt;
        }while(if_use.count(t[k]));
        if_use.insert(t[k]);
        int a = 0,b = 10000000;
        while (b--){
                ++ a;
        }
        return t[k];
}
